import sys
from PyQt5.QtWidgets import QWidget, QVBoxLayout, QTextEdit, QPushButton
from PyQt5.QtCore import Qt, QTimer
import subprocess
import json

class ChatOverlay(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Local AI Assistant")
        self.setGeometry(1200, 100, 400, 500)
        self.setWindowFlags(Qt.WindowStaysOnTopHint | Qt.FramelessWindowHint)
        
        layout = QVBoxLayout()
        self.chat_display = QTextEdit()
        self.chat_display.setReadOnly(True)
        self.input_box = QTextEdit()
        self.input_box.setMaximumHeight(80)
        
        send_btn = QPushButton("Send")
        send_btn.clicked.connect(self.send_message)
        
        layout.addWidget(self.chat_display)
        layout.addWidget(self.input_box)
        layout.addWidget(send_btn)
        self.setLayout(layout)
        
        self.chat_history = []
        
    def send_message(self):
        user_input = self.input_box.toPlainText().strip()
        if not user_input:
            return
            
        self.chat_history.append({"role": "user", "content": user_input})
        self.input_box.clear()
        
        # Call Ollama
        response = self.call_ollama(user_input)
        self.chat_history.append({"role": "assistant", "content": response})
        
        self.update_display()
        
    def call_ollama(self, prompt):
        result = subprocess.run(
            ["ollama", "run", "llama3.1:8b", prompt],
            capture_output=True,
            text=True
        )
        return result.stdout.strip()
    
    def update_display(self):
        text = ""
        for msg in self.chat_history[-10:]:  # Last 10 messages
            text += f"\n{msg['role'].upper()}:\n{msg['content']}\n"
        self.chat_display.setText(text)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    overlay = ChatOverlay()
    overlay.show()
    sys.exit(app.exec_())

class SkillGenerator:
    def __init__(self, agent_core):
        self.agent = agent_core
        self.skills_dir = Path("~/.agent_skills").expanduser()
        self.skills_dir.mkdir(exist_ok=True)
        
    def generate_skill_from_task(self, task_name, task_steps, execution_log):
        """
        When agent completes a complex task, extract a reusable skill.
        """
        # Use LLM to generate Python code for the skill
        prompt = f"""
        Task: {task_name}
        Steps taken: {task_steps}
        Execution log: {execution_log}
        
        Generate a reusable Python function that automates this task.
        Include docstring, type hints, error handling.
        """
        
        skill_code = self.call_ollama(prompt)
        
        # Save skill
        skill_file = self.skills_dir / f"{task_name}_skill.py"
        skill_file.write_text(skill_code)
        
        # Generate metadata
        metadata = {
            "name": task_name,
            "description": "Auto-generated from task execution",
            "created_at": datetime.now().isoformat(),
            "usage_count": 0,
            "success_rate": 1.0,
            "relevant_keywords": self.extract_keywords(task_name)
        }
        
        metadata_file = self.skills_dir / f"{task_name}_meta.json"
        metadata_file.write_text(json.dumps(metadata, indent=2))
        
        return skill_file
    
    def prune_skills(self):
        """Remove low-usage, low-success-rate skills"""
        for meta_file in self.skills_dir.glob("*_meta.json"):
            meta = json.loads(meta_file.read_text())
            if meta["success_rate"] < 0.5 and meta["usage_count"] > 10:
                # Delete skill
                meta_file.unlink()
                meta_file.stem.replace("_meta", "_skill") + ".py"

import speech_recognition as sr
from elevenlabs import client, play

class VoiceIO:
    def __init__(self):
        self.recognizer = sr.Recognizer()
        
    def listen(self):
        """Capture voice input"""
        with sr.Microphone() as source:
            audio = self.recognizer.listen(source)
            try:
                text = self.recognizer.recognize_google(audio)
                return text
            except:
                return None
    
    def speak(self, text):
        """Text to speech via ElevenLabs"""
        audio = client.generate(text=text, voice="Bella")
        play(audio)
